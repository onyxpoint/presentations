Classes and Definitions

  • Building Blocks of Reusable Functionality
  • Classes - Singletons => Can Inherit
  • Defines - Multiple => Cannot Inherit

Classes


class 'ssh' (
  $enable_service => true,
  $port => '22'
){
  file { '/etc/ssh/sshd_config':
    content => template('modules/ssh/sshd_config.erb')
  }
  service { 'sshd':
    enable => $enable_service,
    subscribe => File['/etc/ssh/sshd_config']
  }
}
  

include 'ssh'

Defines


define mkusr (
  $uid
) {
  group { $name:
    gid => $uid
  }

  user { $name:
    uid => $uid,
    gid => $name
  }
}
  

mkusr { 'bob': uid => '1111' }
mkusr { 'alice': uid => '1112' }
mkusr { 'eve': uid => '1337' }
